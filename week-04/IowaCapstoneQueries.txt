-- 1. Create a list of all transactions for your chosen [Category/Vendor] that took place in 
-- the last quarter of 2014, sorted by the total sale amount from highest to lowest. 
-- (Strength: Identifying high-volume peak periods).

SELECT *
FROM sales
WHERE date BETWEEN '2014-10-01' AND '2014-12-31'
	AND vendor ILIKE '%bacardi%'
	ORDER BY total DESC

-- 2. Which transactions for your [Category/Vendor] had a bottle quantity greater than 12? 
-- Display the date, store number, item description, and total amount. 
-- (Strength: Identifying bulk buyers or wholesale-style transactions).

SELECT date, store, description, total
FROM sales
WHERE bottle_qty > 12
	AND vendor ILIKE '%bacardi%'
	ORDER BY 3 DESC

-- 3. Find all products in the products_table whose item_description contains a specific 
-- keyword (e.g., 'Limited', 'Spiced'). What categories do they belong to? 
-- (Opportunity: Identifying niche product variants).


SELECT category_name , item_description , vendor_name
FROM products
WHERE (item_description ILIKE '%limited%' 
	OR
	item_description ILIKE '%spiced%'
	OR
	item_description ILIKE '%reserve%'
	OR
	item_description ILIKE '%rare%'
	OR
	item_description ILIKE '%exclusive%'
	OR
	item_description ILIKE '%special%')
	AND
	vendor_name ILIKE '%bacardi%'
	ORDER BY 1


-- 4. What is the total sales revenue and the average bottle price (btl_price) for 
-- your chosen [Category/Vendor]? 
-- (Strength/Baseline: Establishing the financial footprint).

SELECT 'Bacardi' AS Vendor_Name , 
SUM(total):: money AS Total_Revenue ,
ROUND(AVG(btl_price::numeric) , 2) AS Avg_Bottle_Price
FROM sales
WHERE vendor ILIKE '%bacardi%'

-- 5. How many transactions were recorded for each specific item description within your 
-- chosen [Category]? Which specific product is the most frequently purchased? 
-- (Strength: Identifying your "hero" product).

SELECT description AS item_name , COUNT(item) AS transaction_count
FROM sales
WHERE vendor ILIKE '%bacardi%'
GROUP BY description
ORDER BY COUNT(item) DESC


-- 6. Which store generated the highest total revenue for your [Category/Vendor]? 
-- Which generated the lowest (but still greater than zero)? 
-- (Strength vs. Weakness: Identifying your best and worst retail partners).

--Use this to find the store with highest revenue:
SELECT st.name , SUM(s.total)::money
FROM sales AS s
JOIN stores AS st ON st.store=s.store
WHERE s.vendor ILIKE '%bacardi%'
GROUP BY 1
ORDER BY SUM(s.total) DESC
LIMIT 1

--Use this to find the store with lowest revenue:
SELECT st.name , SUM(s.total)::money
FROM sales AS s
JOIN stores AS st ON st.store=s.store
WHERE s.vendor ILIKE '%bacardi%' 
	AND 2 > 0
GROUP BY 1
ORDER BY SUM(s.total) ASC
LIMIT 1

-- 8. Which stores had total sales revenue for your [Category/Vendor] exceeding $2,000? 
-- (Hint: Use HAVING to filter aggregated store totals). 
-- (Strength: Pinpointing high-performing retail locations).

SELECT st.name , s.store , SUM(s.total)::money AS store_revenue
FROM sales AS s
JOIN stores AS st ON st.store = s.store
WHERE s.vendor ILIKE '%bacardi%'
GROUP BY 1,2
HAVING SUM(s.total) > 2000
ORDER BY 3 DESC

-- 9. Find all sales records where the category_name is either your chosen category 
-- or a closely related competitor category (e.g., 'VODKA 80 PROOF' vs 'IMPORTED VODKA'). 
-- (Threat: Comparing performance against direct substitutes).

SELECT category_name , COUNT(item) , SUM(total)
FROM sales
WHERE vendor ILIKE '%bacardi%'
GROUP BY 1
ORDER BY 2 DESC

--The query above is telling me which categories are Bacardi is strongest.
-- I checked both number of items sold and the amount of revenue generated.
-- Top 3 categories (below) in both metrics are the same categories. 
-- Accordingly, I found my competitors in these categories with the query below.
--"PUERTO RICO & VIRGIN ISLANDS RUM"	64112	$7,666,834.89
--"IMPORTED VODKA"	29785	$6,536,589.38
--"FLAVORED RUM"	27319	$2,694,923.00

SELECT vendor, category_name, SUM(total)::MONEY
FROM sales
WHERE category_name IN ('PUERTO RICO & VIRGIN ISLANDS RUM','IMPORTED VODKA','FLAVORED RUM')
GROUP BY 1,2
ORDER BY 3 DESC

-- My top competitors in these categories are 
-- "Pernod Ricard USA/Austin Nichols" >> "IMPORTED VODKA"
-- "Pernod Ricard USA/Austin Nichols" >> "FLAVORED RUM"
-- "Diageo Americas" >>	"IMPORTED VODKA"
-- "Luxco-St Louis" >> "PUERTO RICO & VIRGIN ISLANDS RUM"
-- Accordingly, the query below is listing all sales data of Bacardi and its competitors.

SELECT vendor , category_name , SUM(total)::MONEY AS revenue,
CASE
    WHEN vendor ILIKE '%Bacardi%' THEN 'Bacardi'
    WHEN vendor ILIKE '%Pernod Ricard%' THEN 'Pernod Ricard'
    WHEN vendor ILIKE '%Diageo%' THEN 'Diageo'
    WHEN vendor ILIKE '%Luxco%' THEN 'Luxco'
END AS vendor_name
FROM sales
WHERE 
(vendor ILIKE '%Bacardi%' OR
	vendor ILIKE '%Pernod Ricard%' OR
	vendor ILIKE '%Diageo%' OR	
	vendor ILIKE '%Luxco%' )
	AND (category_name ILIKE '%IMPORTED VODKA%'OR
	category_name ILIKE '%FLAVORED RUM%'OR
	category_name ILIKE '%PUERTO RICO & VIRGIN ISLANDS RUM%'
	)
GROUP BY 1,2
ORDER BY 3 DESC

-- 10. List all transactions where the state bottle cost was between $10 and $20 for 
-- your [Category/Vendor]. 
-- (Opportunity: Analyzing performance in the "mid-tier" price bracket).


SELECT st.name AS store_name,
       p.item_description,
       s.state_btl_cost::money,
       s.btl_price::money,
       SUM(s.total)::money AS revenue
FROM sales AS s
JOIN stores AS st ON st.store = s.store
JOIN products AS p ON p.item_no = s.item
WHERE s.vendor ILIKE '%bacardi%'
AND s.btl_price::numeric BETWEEN 10 AND 20
GROUP BY 1,2,3,4
ORDER BY 5 DESC

-- The direct short answer to this question doesn't require JOINing the stores,
-- However I joined the stores to see our largest partners in mid-tier bracket.
-- This information can be joined with transaction dates and we can see trends in stores.
-- If we see significant decline in our large partners we can focus on them.
-- I am adding an extra query below to see transaction amounts per store/month

SELECT st.name AS store_name,
       DATE_TRUNC('month', s.date) AS month,
       COUNT(s.item) AS transactions,
       SUM(s.total)::money AS monthly_revenue
FROM sales AS s
JOIN stores AS st ON st.store = s.store
WHERE s.vendor ILIKE '%bacardi%'
AND s.btl_price::numeric BETWEEN 10 AND 20
GROUP BY 1,2
ORDER BY 1,2


-- 11. Write a query that displays each store's total sales for your [Category/Vendor] 
-- along with the store's name and address from the stores_table.

	st.store_address,

FROM sales AS s

	WHERE s.vendor ILIKE '%bacardi%'
GROUP BY 1,2
ORDER BY 4 DESC

-- 12. For each sale in your [Category], display the transaction date, total amount, 
-- and the population of the county where the sale occurred by joining with counties_table. 
-- (Opportunity: Correlating sales volume with population density).

SELECT s.date AS transaction_date,
	c.population,
	c.county,
	SUM(s.total)::money AS total_revenue,
FROM sales AS s
JOIN counties AS c ON c.county = s.county
WHERE s.category_name ILIKE '%FLAVORED RUM%'
GROUP BY 1,2,3
ORDER BY 4 DESC

-- I also wanted to show the revenue per capita in each county without transaction dates.
SELECT c.population,
	c.county,
	SUM(s.total)::money AS total_revenue,
	ROUND((SUM(s.total) / c.population ), 2) AS revenue_per 
FROM sales AS s
JOIN counties AS c ON c.county = s.county
WHERE s.category_name ILIKE '%FLAVORED RUM%'
GROUP BY 1,2
ORDER BY 4 DESC	


-- 13. Write a query that shows total sales for your [Category/Vendor] by county. 
-- Which county generates the most revenue for you? 
-- (Strength: Identifying your geographic stronghold).

SELECT	c.county,
	SUM(s.total)::money AS total_revenue,
	CASE
    WHEN vendor ILIKE '%Bacardi%' THEN 'Bacardi'
	END AS vendor_name
FROM sales as s
JOIN counties as c ON c.county = s.county
WHERE s.vendor ILIKE '%bacardi%'
GROUP BY 1,3
ORDER BY 2 DESC


-- 14. Join the sales_table and products_table to show total revenue for your [Vendor] 
-- alongside the proof and pack size of the items. 
-- (Strength: Determining if higher proof or larger packs drive more revenue).

SELECT p.item_description,
	p.item_no,
	p.pack,
	p.inner_pack,
	p.proof,
	p.bottle_size,
	SUM(s.total)::money AS total_revenue
FROM products AS p
JOIN sales AS s ON s.item = p.item_no
WHERE p.vendor_name ILIKE '%bacardi%'
GROUP BY 1,2,3,4,5,6
ORDER BY 7 DESC

-- 17. Using a subquery, find all transactions for your [Category/Vendor] from stores
-- located in a specific high-growth city (e.g., 'Des Moines') found in the stores_table. 
-- (Opportunity: Drilling down into urban market performance).

SELECT date, total, store_name
FROM (
    SELECT s.date,
           s.county,
           s.total,
           s.vendor,
           st.name AS store_name 
    FROM sales AS s
    JOIN stores AS st ON st.store = s.store
) AS Transactions
WHERE county = 'Polk'
AND vendor ILIKE '%bacardi%'
ORDER BY 1 ASC


-- 18. Which stores had total sales for your [Category/Vendor] that were above the 
-- average store revenue for that same group? (Hint: Use a subquery for the average). 
-- (Strength: Identifying stores that are over-performing).
SELECT name,
	county,
	SUM(total)::money AS store_revenue,
	AVG(avg_revenue)::money AS average_revenue
FROM(SELECT st.name,
	s.county,
	s.total,
	AVG(s.total)OVER() AS avg_revenue 
	FROM sales AS s
	JOIN stores AS st ON st.store=s.store
	WHERE vendor ILIKE '%bacardi%' ) AS avg_rev_table

GROUP BY 1,2
HAVING SUM(total) > AVG(avg_revenue)
ORDER BY 3 DESC
